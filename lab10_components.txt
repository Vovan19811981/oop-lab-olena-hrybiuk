ФАЙЛ: BtnNClick.h (Заголовний файл компонента)

#ifndef BtnNClickH
#define BtnNClickH

#include <SysUtils.hpp>
#include <Classes.hpp>
#include <Controls.hpp>
#include <StdCtrls.hpp>

class PACKAGE TBtnNClick : public TButton
{
private:
    int NClick;  // Змінна для підрахунку кількості натискань
    
protected:
    
public:
    // Конструктор нового компонента
    __fastcall TBtnNClick(TComponent* Owner);
    
    // Віртуальний метод обробки натискання
    DYNAMIC void __fastcall Click(void);
    
    // Метод для отримання кількості натискань
    int __fastcall GetClickCount(void);
    
    // Метод для скидання лічильника
    void __fastcall ResetCounter(void);
    
__published:
};

#endif

ФАЙЛ: BtnNClick.cpp (Реалізація компонента)

#include <vcl.h>
#pragma hdrstop

#include "BtnNClick.h"
#pragma package(smart_init)

//-----------------------------------------------------------------------------
// Перевірка валідності компонента
//-----------------------------------------------------------------------------
static inline void ValidCtrCheck(TBtnNClick *)
{
    new TBtnNClick(NULL);
}

//-----------------------------------------------------------------------------
// КОНСТРУКТОР компонента TBtnNClick
// Ініціалізує лічильник натискань нульовим значенням
//-----------------------------------------------------------------------------
__fastcall TBtnNClick::TBtnNClick(TComponent* Owner)
    : TButton(Owner)
{
    NClick = 0;  // Початкове значення лічильника
}

//-----------------------------------------------------------------------------
// МЕТОД Click
// Обробляє подію натискання кнопки:
// - збільшує лічильник на 1
// - відображає кількість натискань на кнопці
// - викликає батьківський метод Click
//-----------------------------------------------------------------------------
void __fastcall TBtnNClick::Click(void)
{
    // Збільшуємо лічильник
    NClick++;
    
    // Формуємо текст кнопки з кількістю натискань
    Caption = Name + " (" + IntToStr(NClick) + ") клацань";
    
    // Викликаємо успадкований метод Click від TButton
    // Це забезпечує нормальну роботу події OnClick
    TButton::Click();
}

//-----------------------------------------------------------------------------
// МЕТОД GetClickCount
// Повертає поточне значення лічильника натискань
//-----------------------------------------------------------------------------
int __fastcall TBtnNClick::GetClickCount(void)
{
    return NClick;
}

//-----------------------------------------------------------------------------
// МЕТОД ResetCounter
// Скидає лічильник натискань до нуля
//-----------------------------------------------------------------------------
void __fastcall TBtnNClick::ResetCounter(void)
{
    NClick = 0;
    Caption = Name;
}

//-----------------------------------------------------------------------------
// ФУНКЦІЯ Register
// Реєструє компонент на палітрі компонентів
// Розміщує його на сторінці "MyClasses"
//-----------------------------------------------------------------------------
namespace Btnnclick
{
    void __fastcall PACKAGE Register()
    {
        TComponentClass classes[1] = {__classid(TBtnNClick)};
        RegisterComponents("MyClasses", classes, 0);
    }
}

#include <vcl.h>
#pragma hdrstop

#include "BtnNClick.cpp"  // Підключаємо модуль компонента
#include "Unit1.h"

#pragma package(smart_init)
#pragma resource "*.dfm"

TForm1 *Form1;
TBtnNClick *Btn;  // Глобальна змінна для нового компонента

//-----------------------------------------------------------------------------
// КОНСТРУКТОР форми
//-----------------------------------------------------------------------------
__fastcall TForm1::TForm1(TComponent* Owner)
    : TForm(Owner)
{
}

//-----------------------------------------------------------------------------
// FormCreate - Ініціалізація форми
//-----------------------------------------------------------------------------
void __fastcall TForm1::FormCreate(TObject *Sender)
{
    Btn = NULL;
    Label1->Caption = "Натисніть 'Створити нову кнопку'";
}

//-----------------------------------------------------------------------------
// Button1Click - Створення нової кнопки TBtnNClick
//-----------------------------------------------------------------------------
void __fastcall TForm1::Button1Click(TObject *Sender)
{
    if (Btn != NULL)
    {
        ShowMessage("Кнопка вже створена! Спочатку видаліть її.");
        return;
    }
    
    // Створюємо новий екземпляр компонента
    Btn = new TBtnNClick(Form1);
    
    // Встановлюємо батьківський компонент
    Btn->Parent = Form1;
    
    // Задаємо властивості
    Btn->Name = "MyButton";
    Btn->Caption = "MyButton (0) клацань";
    Btn->Left = 20;
    Btn->Top = 150;
    Btn->Width = 200;
    Btn->Height = 40;
    Btn->Font->Size = 10;
    Btn->Font->Style = TFontStyles() << fsBold;
    
    Label1->Caption = "Кнопка створена! Натискайте на неї.";
}

//-----------------------------------------------------------------------------
// Button2Click - Показати кількість натискань
//-----------------------------------------------------------------------------
void __fastcall TForm1::Button2Click(TObject *Sender)
{
    if (Btn == NULL)
    {
        ShowMessage("Спочатку створіть кнопку!");
        return;
    }
    
    int count = Btn->GetClickCount();
    Label1->Caption = "Кількість натискань: " + IntToStr(count);
    ShowMessage("Кнопку натиснуто " + IntToStr(count) + " разів");
}

//-----------------------------------------------------------------------------
// Button3Click - Скинути лічильник
//-----------------------------------------------------------------------------
void __fastcall TForm1::Button3Click(TObject *Sender)
{
    if (Btn == NULL)
    {
        ShowMessage("Спочатку створіть кнопку!");
        return;
    }
    
    Btn->ResetCounter();
    Label1->Caption = "Лічильник скинуто";
}

//-----------------------------------------------------------------------------
// Button4Click - Видалити кнопку
//-----------------------------------------------------------------------------
void __fastcall TForm1::Button4Click(TObject *Sender)
{
    if (Btn == NULL)
    {
        ShowMessage("Кнопка ще не створена!");
        return;
    }
    
    delete Btn;
    Btn = NULL;
    Label1->Caption = "Кнопка видалена";
}

#ifndef SGH
#define SGH

#include <SysUtils.hpp>
#include <Classes.hpp>
#include <Controls.hpp>
#include <Grids.hpp>
#include <Graphics.hpp>

class PACKAGE TSG : public TStringGrid
{
private:
    TColor OriginalColor;      // Збереження оригінального кольору
    int OriginalFontSize;      // Збереження оригінального розміру шрифту
    int LastClickedCol;        // Остання клацнута колонка
    int LastClickedRow;        // Остання клацнутий рядок
    bool FontEnlarged;         // Чи збільшений шрифт
    
protected:
    // Обробка натискання клавіш
    void __fastcall KeyDown(Word &Key, TShiftState Shift);
    
    // Обробка клацання мишею

                             int X, int Y);
    
public:
    // Конструктор
    __fastcall TSG(TComponent* Owner);
    
    // Метод заповнення випадковими числами
    void __fastcall FillRandom(void);
    
    // Метод очищення всіх комірок
    void __fastcall ClearAll(void);
    
    // Метод збільшення шрифту в комірці
    void __fastcall EnlargeFont(int Col, int Row);
    
    // Метод повернення шрифту до нормального розміру
    void __fastcall RestoreFont(int Col, int Row);
    
__published:
};

#endif

ФАЙЛ: SG.cpp (Реалізація компонента)

#include <vcl.h>
#pragma hdrstop

#include "SG.h"
#include <stdlib.h>
#include <time.h>

#pragma package(smart_init)

//-----------------------------------------------------------------------------
// Перевірка валідності компонента
//-----------------------------------------------------------------------------
static inline void ValidCtrCheck(TSG *)
{
    new TSG(NULL);
}

//-----------------------------------------------------------------------------
// КОНСТРУКТОР компонента TSG
// Ініціалізує початкові значення
//-----------------------------------------------------------------------------
__fastcall TSG::TSG(TComponent* Owner)
    : TStringGrid(Owner)
{
    // Ініціалізація генератора випадкових чисел
    randomize();
    
    // Збереження оригінальних властивостей
    OriginalColor = clWindow;
    OriginalFontSize = Font->Size;
    
    // Ініціалізація змінних для відстеження клацань
    LastClickedCol = -1;
    LastClickedRow = -1;
    FontEnlarged = false;
    
    // Встановлення властивостей за замовчуванням
    DefaultColWidth = 60;
    DefaultRowHeight = 24;
    Options = Options << goEditing << goTabs;
}

//-----------------------------------------------------------------------------
// МЕТОД KeyDown
// Обробляє натискання клавіш:
// F11 - заповнення випадковими числами
// ESC - очищення всіх комірок
//-----------------------------------------------------------------------------
void __fastcall TSG::KeyDown(Word &Key, TShiftState Shift)
{
    // Обробка клавіші F11 - заповнення випадковими числами
    if (Key == VK_F11)
    {
        FillRandom();
        Key = 0;  // Скасувати подальшу обробку клавіші
    }
    // Обробка клавіші ESC - очищення всіх комірок
    else if (Key == VK_ESCAPE)
    {
        ClearAll();
        Key = 0;
    }
    else
    {
        // Виклик батьківського методу для інших клавіш
        TStringGrid::KeyDown(Key, Shift);
    }
}

//-----------------------------------------------------------------------------
// МЕТОД MouseDown
// Обробляє клацання мишею:
// - Збільшує шрифт у клацнутій комірці
// - Або повертає до нормального розміру при повторному клацанні
//-----------------------------------------------------------------------------
void __fastcall TSG::MouseDown(TMouseButton Button, TShiftState Shift,
                               int X, int Y)
{
    int ACol, ARow;
    
    // Виклик батьківського методу
    TStringGrid::MouseDown(Button, Shift, X, Y);
    
    // Обробка лише лівої кнопки миші
    if (Button == mbLeft)
    {
        // Визначаємо координати комірки
        MouseToCell(X, Y, ACol, ARow);
        
        // Перевіряємо, чи це не фіксована комірка
        if ((ACol >= FixedCols) && (ARow >= FixedRows))
        {
            // Якщо це та сама комірка, що була клацнута раніше
            if ((ACol == LastClickedCol) && (ARow == LastClickedRow) && FontEnlarged)
            {
                // Повертаємо шрифт до нормального розміру
                RestoreFont(ACol, ARow);
                FontEnlarged = false;
            }
            else
            {
                // Якщо була клацнута інша комірка раніше, повертаємо її шрифт
                if (FontEnlarged && (LastClickedCol >= 0) && (LastClickedRow >= 0))
                {
                    RestoreFont(LastClickedCol, LastClickedRow);
                }
                
                // Збільшуємо шрифт у новій комірці
                EnlargeFont(ACol, ARow);
                
                // Зберігаємо координати
                LastClickedCol = ACol;
                LastClickedRow = ARow;
                FontEnlarged = true;
            }
        }
    }
}

//-----------------------------------------------------------------------------
// МЕТОД FillRandom
// Заповнює всі комірки випадковими числами від 0 до 20
//-----------------------------------------------------------------------------
void __fastcall TSG::FillRandom(void)
{
    for (int i = FixedCols; i < ColCount; i++)
    {
        for (int j = FixedRows; j < RowCount; j++)
        {
            int randomNum = random(21);  // Випадкове число від 0 до 20
            Cells[i][j] = IntToStr(randomNum);
        }
    }
    
    // Скидаємо інформацію про збільшений шрифт
    if (FontEnlarged && (LastClickedCol >= 0) && (LastClickedRow >= 0))
    {
        RestoreFont(LastClickedCol, LastClickedRow);
        FontEnlarged = false;
    }
}

//-----------------------------------------------------------------------------
// МЕТОД ClearAll
// Очищує всі комірки таблиці
//-----------------------------------------------------------------------------
void __fastcall TSG::ClearAll(void)
{
    for (int i = FixedCols; i < ColCount; i++)
    {
        for (int j = FixedRows; j < RowCount; j++)
        {
            Cells[i][j] = "";
        }
    }
    
    // Скидаємо інформацію про збільшений шрифт
    if (FontEnlarged && (LastClickedCol >= 0) && (LastClickedRow >= 0))
    {
        RestoreFont(LastClickedCol, LastClickedRow);
        FontEnlarged = false;
    }
}

//-----------------------------------------------------------------------------
// МЕТОД EnlargeFont
// Збільшує розмір шрифту в заданій комірці вдвічі
//-----------------------------------------------------------------------------
void __fastcall TSG::EnlargeFont(int Col, int Row)
{
    // Зберігаємо оригінальний розмір, якщо це перше збільшення
    if (!FontEnlarged)
    {
        OriginalFontSize = Font->Size;
    }
    
    // Встановлюємо фокус на комірку
    this->Col = Col;
    this->Row = Row;
    
    // Збільшуємо розмір шрифту вдвічі
    Canvas->Font->Size = OriginalFontSize * 2;
    
    // Перемальовуємо комірку
    InvalidateCell(Col, Row);
    
    // Оновлюємо відображення
    Refresh();
}

//-----------------------------------------------------------------------------
// МЕТОД RestoreFont
// Повертає розмір шрифту в заданій комірці до початкового значення
//-----------------------------------------------------------------------------
void __fastcall TSG::RestoreFont(int Col, int Row)
{
    // Встановлюємо фокус на комірку
    this->Col = Col;
    this->Row = Row;
    
    // Повертаємо оригінальний розмір шрифту
    Canvas->Font->Size = OriginalFontSize;
    
    // Перемальовуємо комірку
    InvalidateCell(Col, Row);
    
    // Оновлюємо відображення
    Refresh();
}

//-----------------------------------------------------------------------------
// ФУНКЦІЯ Register
// Реєструє компонент на палітрі компонентів
// Розміщує його на сторінці "Samples"
//-----------------------------------------------------------------------------
namespace Sg
{
    void __fastcall PACKAGE Register()
    {
        TComponentClass classes[1] = {__classid(TSG)};
        RegisterComponents("Samples", classes, 0);
    }
}

ОСОБЛИВОСТІ РЕАЛІЗАЦІЇ

================================================================================

КОМПОНЕНТ TBtnNClick:

✓ Успадковується від TButton
✓ Додано приватне поле NClick для підрахунку
✓ Перекрито метод Click() для обробки натискань
✓ Додано методи GetClickCount() та ResetCounter()
✓ Автоматичне оновлення підпису кнопки

КОМПОНЕНТ TSG:

✓ Успадковується від TStringGrid
✓ Обробка клавіші F11 для заповнення випадковими числами (0-20)
✓ Обробка клавіші ESC для очищення всіх комірок
✓ Обробка клацання мишею для зміни розміру шрифту
✓ Запам'ятовування останньої клацнутої комірки
✓ Автоматичне відновлення шрифту при повторному клацанні

================================================================================

ПРИКЛАДИ ВИКОРИСТАННЯ

================================================================================

ПРИКЛАД 1: Використання TBtnNClick в коді

TBtnNClick *MyBtn = new TBtnNClick(Form1);
MyBtn->Parent = Form1;
MyBtn->Caption = "Клацни мене!";
MyBtn->Left = 50;
MyBtn->Top = 50;

// Отримання кількості натискань
int count = MyBtn->GetClickCount();
ShowMessage("Натиснуто: " + IntToStr(count));

// Скидання лічильника
MyBtn->ResetCounter();

ПРИКЛАД 2: Використання TSG в коді

TSG *Grid = new TSG(Form1);
Grid->Parent = Form1;
Grid->RowCount = 5;
Grid->ColCount = 5;
Grid->Left = 10;
Grid->Top = 10;

// Заповнення випадковими числами програмно
Grid->FillRandom();

// Очищення всіх комірок програмно
Grid->ClearAll();

КОНТРОЛЬНІ ПИТАННЯ ТА ВІДПОВІДІ

================================================================================

1. Що означає специфікатор DYNAMIC?
   Відповідь: DYNAMIC вказує на те, що метод є динамічним (віртуальним).
   Він може бути перевизначений у класах-нащадках. Використовується для
   реалізації поліморфізму.

2. Які елементі є основними у класі компонентів?
   Відповідь: Властивості (Properties), методи (Methods), події (Events).

3. Який клас є базовим для усіх візуальних компонентів?
   Відповідь: TControl

4. Який клас є базовим для усіх віконних компонентів?
   Відповідь: TWinControl

5. Який клас є базовим для графічних компонентів?
   Відповідь: TGraphicControl

6. Назвіть методи класу TControl та їхнє призначення.
   Відповідь:
   - Show() - відображення компонента
   - Hide() - приховування компонента
   - Invalidate() - перемальовування компонента
   - Update() - оновлення компонента
   - SetBounds() - встановлення розміру та позиції

7. Назвіть методи класу TWinControl та їхнє призначення.
   Відповідь:
   - SetFocus() - встановлення фокусу
   - CanFocus() - перевірка можливості отримання фокусу
   - Focused() - перевірка наявності фокусу

8. Назвіть послідовність процесу створення нового компонента?
   Відповідь:
   1) Обирання батьківського компонента
   2) Створення модуля нового компонента
   3) Тестування компонента
   4) Інсталяція компонента до пакета

9. Для чого потрібна інсталяція компонента?
   Відповідь: Інсталяція додає компонент на палітру компонентів, що
   дозволяє використовувати його візуально в Design Time, просто
   перетягуючи на форму, як стандартні компоненти.
